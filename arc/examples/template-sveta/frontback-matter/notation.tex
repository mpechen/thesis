\begin{notations}
\notation{$:=$}{equals by definition}
\notation{$\hookrightarrow$}{compact embedding}
\notation{$\equiv$}{logical equivalence}
\notation{$\forall$}{for all}
\notation{$a \cdot b$}{scalar product of vectors}
%
\notation{$\Nd$}{space of natural numbers}
\notation{$\Real$}{space of real numbers}
\notation{$\Rd$}{space of real $d$-vectors}
\notation{$\Omega$}{open bounded connected domain in $\Rd$ with Lipschitz}
\notation{}{continuous boundary}
\notation{$\overline{\Omega}$}{closure of $\Omega$}
\notation{$ \partial \Omega$}{Lipschitz continuous boundary of $\Omega$}
\notation{$\Gamma$}{part of $ \partial \Omega$ such that $\meas_{d - 1} \Gamma > 0$}
\notation{$Q_T$}{space-time cylinder $Q_T:= \Omega \times (0, T)$, where $T$ is given time}
\notation{$S_T$}{lateral surface of $Q_T$, i.e., $S_T := \partial \Omega \times [0, T]$}
\notation{$\diam {\Omega}$}{diameter of the set $\Omega$}
\notation{$\meas {\Omega}$}{Lebesgue measure of the set $\Omega$}
%
\notation{$D^\alpha v$}{derivative of order $|\alpha|$}
\notation{$\C{k}(\Omega)$}{space of $k$-times differentiable scalar-valued functions}
\notation{$\Co{k}(\Omega)$}{subspace of $C^k(\Omega)$ that contains functions with compact}
\notation{ }{ support in $\Omega$}
\notation{$\Co{\infty}(\Omega)$}{space of smooth functions with compact support in $\Omega$}
%\notation{$P^{\, k}(\Omega)$}{set of polynomial functions defined in $\Omega\subset\Rd$}
\notation{$\L{p}(\Omega)$}{space of scalar-valued functions in $\Omega$ summable with power $p$}
\notation{$\L{p}(\Omega,\Rd)$}{space of vector-valued functions with components summable}
\notation{$                 $}{with power $p$ in $\Omega$}
\notation{$X$}{Banach space}
\notation{$V$}{Hilbert space}
\notation{$V^*$}{space dual to $V$}
\notation{$\W{l, p}{}(\Omega)$}{Sobolev space of functions $w$ summable with power $p$ and} 
\notation{$$}{possessing derivatives $D^\alpha w \in \L{p}(\Omega)$, $|\alpha| \leq l$}
\notation{$\H{l}(\Omega)$}{Sobolev space $\W{l, p}{}$ with $p = 2$}
\notation{$\Ho{l}(\Omega)$}{subspace of $\H{l}(\Omega)$ formed by functions vanishing on $\Gamma$}
\notation{$\H{-1}(\Omega)$}{space dual to $\Ho{1}(\Omega)$}
\notation{$H(\Omega,\dvrg)$}{subspace of $\L{2}(\Omega,\Rd)$ that contains vector-valued functions} \notation{$                $}{with square-summable divergence}
%
\notation{$\| \cdot \|_X$}{norm in space $X$}
\notation{$\| \cdot \|$}{norm in $\L{2}(\Omega)$}
%\notation{$\| \cdot \|_{\L{p}}$}{norm in space of function $\L{p}(\Omega)$}
%\notation{$\| \cdot \|_{\L{\infty}}$}{norm in space of essentially bounded functions in $\L{\infty}(\Omega)$}
%\notation{$\| \cdot \|_{\H{1}}$}{norm in space $\W{l}{p}$}
%\notation{$\| \cdot \|_{\W{l, p}{}}$}{norm in space $\W{l}{p}$}
\notation{${\mid\!\mid\!\mid \cdot \mid\!\mid\!\mid}$}{energy norm}
\notation{$\| w \|_A$}{a weighed norm in $\L{2}(\Omega)$, i.e., $\big( \IntO A w \cdot w \dx \big)^{\rfrac{1}{2}}$}
\notation{$\| w \|_{A^{-1}}$}{$\big( \IntO A^{-1} w \cdot w \dx \big)^{\rfrac{1}{2}}$}
%
\notation{$w_{t}$}{partial derivative with respect to time coordinate}
\notation{$w_{,i}$}{partial derivative with respect to space $i^{\rm th}$ coordinate}
\notation{$\nabla$}{gradient of a scalar-valued function $\nabla w = \left(w_{,1}, \ldots, w_{,d}\right)$}
\notation{$\Delta$}{Laplace operator $\Delta w := \dvrg \nabla w$}
\notation{$\dvrg$}{divergence of a vector-valued function $\dvrg \, w = \Sum_{i = 1}^{d} w_{i,i}$}
\notation{$\{ w \}_{\Omega}$}{mean value of $w$ on $\Omega$, i.e., $\,{  \{  u \} }_{\Omega} := \tfrac{1}{|\Omega|} \Int_{\Omega} w \dx$}
\notation{$\uppi$}{projection operator}
%
%\notation{$\Indicator$}{error indicator}
\notation{$e$}{error}
\notation{$\genmaj{}$}{majorant functional}
\notation{$\genmij{}$}{minorant functional}
\notation{$\Ieff$}{efficiency index $\Ieff := \tfrac{\genmaj{}}{\mid\!\mid\!\mid e \mid\!\mid\!\mid}$}
\notation{$\Marker$}{marker}
\notation{$\Marker_{\rm AVR}$}{marker defined by the level of the average error}
\notation{$\Marker_{\theta}$}{marker with bulk parameter $\theta$}
\notation{${\rm P}_{k}$}{Lagrangian finite element space of order k}
\notation{${\rm RT}_{k}$}{Raviart-Tomas finite element space of order k}



\end{notations}
